        // Convertir un fichier en base64

        function convertToBase64(file) {

            return new Promise((resolve, reject) => {

                const reader = new FileReader();

                reader.onload = () => {

                    const base64 = reader.result.split(',')[1];

                    resolve({

                        filename: file.name,

                        mimeType: file.type,

                        base64: base64

                    });

                };

                reader.onerror = reject;

                reader.readAsDataURL(file);

            });

        }
