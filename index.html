<script>

/* ================= CONFIG ================= */

const GOOGLE_SCRIPT_URL = "https://script.google.com/macros/s/AKfycbwDJbcdk8eeiZswIhf62xkJZx4cAwHCF8H7jSaxxKjzJw5h-fW2EkKbjbZjjZoPkOVH/exec";

const STRIPE_19 = "https://buy.stripe.com/9B68wQ2EG1zx4Lvf0C5kk00";

const STRIPE_59 = "https://buy.stripe.com/3cI00kgvwceb91LcSu5kk01";


/* ================= PHOTOS ================= */

let photosData = [];

const photosInput = document.getElementById("photos");

const preview = document.getElementById("preview");

document.getElementById("photoBox").onclick = () => photosInput.click();


photosInput.onchange = async e => {

  for (let f of e.target.files) {

    if (photosData.length >= 4) return;

    const data = await compress(f);

    photosData.push({ name:f.name, data });

  }

  renderPhotos();

};


function renderPhotos(){

  preview.innerHTML = photosData.map(p => `<img src="${p.data}">`).join("");

}


async function compress(file){

  const img = new Image();

  img.src = URL.createObjectURL(file);

  await img.decode();

  const c = document.createElement("canvas");

  const r = Math.min(1600/img.width,1600/img.height,1);

  c.width = img.width*r; c.height = img.height*r;

  c.getContext("2d").drawImage(img,0,0,c.width,c.height);

  return c.toDataURL("image/jpeg",0.8);

}


/* ================= ETAPES ================= */

const stepsDiv = document.getElementById("steps");

function addStep(){

  const d = document.createElement("div");

  d.innerHTML = `

    <input required placeholder="Lieu"><br>

    <textarea required placeholder="Description"></textarea><br><br>`;

  stepsDiv.appendChild(d);

}

document.getElementById("addStep").onclick = addStep;

addStep(); addStep(); addStep();


/* ================= SUBMIT ================= */

document.getElementById("form").onsubmit = async e => {

  e.preventDefault();

  const error = document.getElementById("error");

  error.textContent = "";


  if (photosData.length === 0) return error.textContent = "Ajoute au moins une photo";

  if (stepsDiv.children.length < 3) return error.textContent = "Ajoute au moins 3 étapes";


  const fd = new FormData(e.target);

  const data = {

    prenom1: fd.get("prenom1"),

    prenom2: fd.get("prenom2"),

    email: fd.get("email"),

    optionCadre: document.getElementById("frame").checked,

    prixTotal: document.getElementById("frame").checked ? "59€" : "19€",

    photos: photosData,

    etapes: [...stepsDiv.children].map((s,i)=>({

      numero:i+1,

      lieu:s.querySelector("input").value,

      description:s.querySelector("textarea").value

    }))

  };


  document.getElementById("submit").disabled = true;


  await fetch(GOOGLE_SCRIPT_URL, {

    method:"POST",

    headers:{ "Content-Type":"text/plain;charset=utf-8" },

    body:JSON.stringify(data)

  });


  /* ✅ REDIRECTION STRIPE (SIMPLE & FIABLE) */

  const stripeUrl = document.getElementById("frame").checked ? STRIPE_59 : STRIPE_19;

  window.location.href = stripeUrl + "?prefilled_email=" + encodeURIComponent(data.email);

};
